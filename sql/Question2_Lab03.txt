-- String manipulation using LTRIM and RTRIM
SELECT 
    LTRIM(course_no, 'MIS') AS trimmed_left,
    RTRIM(course_no, '101') AS trimmed_right
FROM course
WHERE course_name LIKE '%Intro%';
